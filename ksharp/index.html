K# Documentation


Variables

1. Making a variable

Syntax:

var var_name = value;

Examples:

var a = 1;
var b = 2;
print(a);
print(b);
# This prints 12 on console, see "print and println".

2. Setting a variable

K# is dynamic-typed, so unlike C#, you could set a string to an int.

Syntax:

a = value; #Assignment
a += value; #Add
a -= value; #Subtract 
a *= value; #Multiply 
...
It also applies to every python operator that has "=" (except :=)

Examples:

var a = 1;
print(a); #1
a += 1;
print(a); #2

3. Type functions

This has all builtin type functions for functions that return None (in-place)

Syntax:

varname append item; #varname.append(item) in py
varname extend iterable; #varname.extend(iterable)
varname remove item; #varname.remove(item)
varname sort; #varname.sort()
varname reverse; #varname.reverse()
varname clear; #varname.clear()
varname insert (index, value); #varname.insert(index, value)
varname add item; #varname.add(item)
varname update item; #varname.update(item)
varname discard item; #varname.discard(item)
varname intersection_update iterable; #varname.intersection_update(iterable)

Examples:

var a = [];
var b = input("Enter a value: "); #input 1
a append 'b';    #for tostring conversion on k#
print(a); #['1'] bcz user input
b = input("Enter a value: "); #input 2
a append b;    #for eval on k#
print(a); #['1', 2]
b = input("Enter a value: "); #input a
a append b;    #error: "a" is not defined

Functions

1. Declaring a function

Syntax:

func funcname() {func_data;}

func funcname(paramname) {func_data;}

func funcname(paramname=defaultval) {func_data;}

func funcname(paramname, anotherparam) {func_data;}

func funcname(paramname, anotherparam=defaultval) {func_data;}

func funcname(paramname, anotherparam=defaultval, anotherparam2=defaultval2) {func_data;}

Example (Area calculator):

func calculatearea(width, height) {
    var a = height * width;
    return a;
}
var w = int(input("Enter width: "));
var h = int(input("Enter height: "));
var area = calculatearea(w, h);
print("Area: ");
println(area);

2. Calling functions

Syntax:

funcname();

funcname(val);

funcname(val1, val2);

funcname(val1, val2, val3); #use many params as you want

Example: See "Area calculator"

3. return

Syntax:

return value;

Example: See "Area calculator"

4. Inheritable functions

This could be useful for class-like structs (no classes in K#) (1.4 has classes now)

Syntax

func funcname : anotherfuncname {code}

func funcname : anotherfuncname(additionalparams) {code}

Example

func nums(a, b) {
println("Value 1: " + str(a));
println("Value 2: " + str(b));
}
func add : nums {
    println("Sum: " + str(a + b));
}
add(1, 2);
# -> Value 1: 1
# Value 2: 2
# Sum: 3

Classes

1. Defining a class

For a base class:

class mybaseclass {
    mycode;
}

For an inherited class:

class myclass : mybaseclass {
    fullcode;
}

2. Using a class

var:

var a = new myclass;
a.mymethod();
a.myvar = 1;

self

class myclass {
    var a = 1;
    func mymethod : self.mymethod {  # do the same of a function
        println(self.a);   # ref-ing a variable in root of myclass
        self.a += 1;
    }
}

Control flows

1. if / else

Syntax:

if (condition) {codegoeshere;}

if (condition) {codegoeshere;} else {anothercodegoeshere;}

Examples:

var a = 1;
if (a == 12) {
    println("Hi12");
}
if (a > 12) {
    print("Hi");
}
else {
    print("Hi2");
}

2. while

Syntax:

while (condition) {codegoeshere;}

Example:

var a = 2;
while (a == 2) {
    print(a);
} # forever print 2

3. for

Syntax:

for (newvarname, iterable) {codegoeshere;}

Example:

var a = [1, 2, 3, 4, 5];
for (b, a) {
    print(b);
} # 12345

4. until

Syntax:

until (condition) {codegoeshere;}

Example:

var a = 2;
until (a != 2) {
print(a);
a += 1;
} # prints 2 once

Modules

1. Creating them

To create a module, first create a file with extension .kshp, then write the module.

2. Using them

Syntax:

using "path/to/module";

Then use the module by functions (without "modulename.functionname") or vars.

Basically, filedata = module + mainprog

Print and Println

So basically, print(text) is print(text, end='') in python

println(text) however is print(text) in python

Pointers

1. Getting a pointer

func main(args) {
    var a = 1;
    var pointer = &a;
    print(pointer); #id
}

2. Getting a value from a pointer

func main(args) {
    var a = 1;
    var pointer = &a;
    println(pointer); #id
    var myval = *pointer;
    println(myval); # 1
}

3. Changing a variable based on a pointer variable

func main(args) {
    var a = 1;
    var pointer = &a;
    println(pointer); #id
    var myval = *pointer;
    println(myval); # 1
    *pointer += 1;
    println(a); # 2
}

References

func change_var(:ref) {
    :ref += 1;
}
func main(args) {
    var a = 1;
    change_var(a);
    print(a); # 2
}

DLL modules

We're using user32 as the dll file here. Available on System32.

1. Initialize the module

extern "user32.dll" as user32;

2. Make function headers (MessageBoxW)

extern "user32.dll" as user32;
dllfunc uint user32.MessageBoxW(void, wchar, wchar, uint);

3. Call the function

user32.MessageBoxW(0, "hello", "testing", 0);

4 (optional). Store in a variable

var a = user32.MessageBoxW(0, "hello", "testing", 0); # 1
